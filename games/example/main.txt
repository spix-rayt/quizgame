# Строки начинающиеся с символа # игнорируются программой

# раунд состоит из категорий. категории состоят из вопросов. вопросы состоят текста вопроса и ответа на него
# Строки начинающиеся с символа + означают блок информации. 
# Например, описание раунда начинается после строки +раунд. чтобы начать описание ещё одного раунда, нужно написать ещё одну строку +раунд.
# После "+категория " пишется название категории
# После "+вопрос" пишется текст вопроса

# параметры и настройки записываются через равно. слева от равно записывается имя параметра, справа от равно записывается значение параметра.
# смысл параметра зависит от блока, в котором он находится и от названия параметра.
# примеры параметров с выдуманными значениями:

# ответ в вопросе:
# ответ = 1

# изображение в вопросе:
# изображение = kek.jpg

# аудио в вопросе:
# аудио = fbi_open_up.mp3

# кот в мешке:
# кот = Крокодилы за 300




# Можно использовать отступы, для улучшения читаемости.

+раунд
    +категория Цвета
        +вопрос Какого цвета жёлтый фломастер?
            ответ = Жёлтого
        +вопрос Сколько цветов у радуги?
            ответ = 6 или 7
        +вопрос
            изображение = Turquoise.jpg
            ответ = Бирюзовый
        +вопрос Какого цвета вода?
            ответ = голубого
        +вопрос
            изображение = TheDress.jpg
            ответ = Черный и синий
    +категория Математика
        +вопрос 2+2
            ответ = 4
        +вопрос Варит отлично твоя голова: пять плюс один получается...
            ответ = шесть
        +вопрос Может ли при делении получиться ноль?
            ответ = да
        +вопрос Наибольшее трехзначное число
            ответ = 999
        +вопрос Чему равно произведение всех чисел?
            ответ = 0
    +категория Мемы звуки
        +вопрос
            аудио = quack_5.mp3
            ответ = кря
        +вопрос Кто дружит с Крокодилом геной?
            настоящая цена = 300
            кот = Крокодилы за 300
            ответ = Чебурашка
